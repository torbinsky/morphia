64      /**
65       * Get an array of all fields declared in the supplied class, and all its
66       * superclasses (except java.lang.Object).
67       *
68       * @param type              the class for which we want to retrieve the Fields
69       * @param returnFinalFields specifies whether to return final fields
70       * @return an array of all declared and inherited fields
71       */
72      public static Field[] getDeclaredAndInheritedFields(final Class type, final boolean returnFinalFields) {
73          List<Field> allFields = new ArrayList<Field>();
74          allFields.addAll(getValidFields(type.getDeclaredFields(), returnFinalFields));
75          Class parent = type.getSuperclass();
76          while ((parent != null) && (parent != Object.class)) {
77              allFields.addAll(getValidFields(parent.getDeclaredFields(), returnFinalFields));
78              parent = parent.getSuperclass();
79          }
80          return allFields.toArray(new Field[allFields.size()]);
81      }

94      protected static List<Method> getDeclaredAndInheritedMethods(final Class type, List<Method> methods) {
95          if ((type == null) || (type == Object.class)) {
96              return methods;
97          }
98          if (methods == null) {
99              methods = new ArrayList<Method>();
100         }
101 
102         Class parent = type.getSuperclass();
103         methods = getDeclaredAndInheritedMethods(parent, methods);
104 
105         for (Method m : type.getDeclaredMethods()) {
106             if (!Modifier.isStatic(m.getModifiers())) {
107                 methods.add(m);
108             }
109         }
110 
111         return methods;
112     }

114     public static List<Field> getValidFields(final Field[] fields, final boolean returnFinalFields) {
115         List<Field> validFields = new ArrayList<Field>();
116         // we ignore static and final fields
117         for (Field field : fields) {
118             if (!Modifier.isStatic(field.getModifiers())
119                     && (returnFinalFields || !Modifier.isFinal(field.getModifiers()))) {
120                 validFields.add(field);
121             }
122         }
123         return validFields;
124     }

323      *
324      * @param field the field
325      * @param c     the class to check against
326      * @return true if the field is parameterized and c is the class that
327      *         parameterizes the field, or is an interface that the
328      *         parameterized class implements, else false
329      */
330     public static boolean isFieldParameterizedWithClass(final Field field, final Class c) {
331         if (field.getGenericType() instanceof ParameterizedType) {
332             ParameterizedType ptype = (ParameterizedType) field.getGenericType();
333             for (Type type : ptype.getActualTypeArguments()) {
334                 if (type == c) {
335                     return true;
336                 }
337                 if (c.isInterface() && implementsInterface((Class) type, c)) {
338                     return true;
339                 }
340             }
341         }
342         return false;
343     }

373     /**
374      * Returns a list of the found Annotations hierarchically.
375      * <p/>
376      * Returns the given class, it's super-classes (and their interfaces),
377      * and finally the interfaces of the given class, in that order.
378      *
379      * @return An ArrayList of <T>.
380      */
381     public static <T> ArrayList<T> getAnnotations(final Class c, final Class<T> annClass) {
382         ArrayList<T> found = new ArrayList<T>();
383         // TODO isn't that actually breaking the contract of @Inherited?
384         if (c.isAnnotationPresent(annClass)) {
385             found.add((T) c.getAnnotation(annClass));
386         }
387 
388         Class parent = c.getSuperclass();
389         while ((parent != null) && (parent != Object.class)) {
390             if (parent.isAnnotationPresent(annClass)) {
391                 found.add((T) parent.getAnnotation(annClass));
392             }
393 
394             // ...and interfaces that the superclass implements
395             for (Class interfaceClass : parent.getInterfaces()) {
396                 if (interfaceClass.isAnnotationPresent(annClass)) {
397                     found.add((T) interfaceClass.getAnnotation(annClass));
398                 }
399             }
400 
401             parent = parent.getSuperclass();
402         }
403 
404         // ...and all implemented interfaces
405         for (Class interfaceClass : c.getInterfaces()) {
406             if (interfaceClass.isAnnotationPresent(annClass)) {
407                 found.add((T) interfaceClass.getAnnotation(annClass));
408             }
409         }
410         return found;
411     }

429     public static Set<Class<?>> getFromDirectory(final File directory, final String packageName)
430             throws ClassNotFoundException {
431         Set<Class<?>> classes = new HashSet<Class<?>>();
432         if (directory.exists()) {
433             for (String file : directory.list()) {
434                 if (file.endsWith(".class")) {
435                     String name = packageName + '.' + stripFilenameExtension(file);
436                     Class<?> clazz = Class.forName(name);
437                     classes.add(clazz);
438                 }
439             }
440         }
441         return classes;
442     }

444     public static Set<Class<?>> getFromJARFile(final String jar, final String packageName) throws IOException,
445             FileNotFoundException, ClassNotFoundException {
446         Set<Class<?>> classes = new HashSet<Class<?>>();
447         JarInputStream jarFile = new JarInputStream(new FileInputStream(jar));
448         JarEntry jarEntry;
449         do {
450             jarEntry = jarFile.getNextJarEntry();
451             if (jarEntry != null) {
452                 String className = jarEntry.getName();
453                 if (className.endsWith(".class")) {
454                     className = stripFilenameExtension(className);
455                     if (className.startsWith(packageName)) {
456                         classes.add(Class.forName(className.replace('/', '.')));
457                     }
458                 }
459             }
460         }
461         while (jarEntry != null);
462         return classes;
463     }

470     public static Set<Class<?>> getClasses(final ClassLoader loader, final String packageName) throws IOException,
471             ClassNotFoundException {
472         Set<Class<?>> classes = new HashSet<Class<?>>();
473         String path = packageName.replace('.', '/');
474         Enumeration<URL> resources = loader.getResources(path);
475         if (resources != null) {
476             while (resources.hasMoreElements()) {
477                 String filePath = resources.nextElement().getFile();
478                 // WINDOWS HACK
479                 if (filePath.indexOf("%20") > 0)
480                     filePath = filePath.replaceAll("%20", " ");
481                 // # in the jar name
482                 if (filePath.indexOf("%23") > 0)
483                     filePath = filePath.replaceAll("%23", "#");
484 
485                 if (filePath != null) {
486                     if ((filePath.indexOf("!") > 0) & (filePath.indexOf(".jar") > 0)) {
487                         String jarPath = filePath.substring(0, filePath.indexOf("!")).substring(
488                                 filePath.indexOf(":") + 1);
489                         // WINDOWS HACK
490                         if (jarPath.indexOf(":") >= 0) {
491                             jarPath = jarPath.substring(1);
492                         }
493                         classes.addAll(getFromJARFile(jarPath, path));
494                     } else {
495                         classes.addAll(getFromDirectory(new File(filePath), packageName));
496                     }
497                 }
498             }
499         }
500         return classes;
501     }

608             } else {
609                 return null;
610             }
611         } else {
612             return null;
613         }

616     /**
617      * Get the actual type arguments a child class has used to extend a generic
618      * base class.
619      *
620      * @param baseClass  the base class
621      * @param childClass the child class
622      * @return a list of the raw classes for the actual type arguments.
623      */
624     public static <T> List<Class<?>> getTypeArguments(Class<T> baseClass, Class<? extends T> childClass) {
625         Map<Type, Type> resolvedTypes = new HashMap<Type, Type>();
626         Type type = childClass;
627         // start walking up the inheritance hierarchy until we hit baseClass
628         while (!getClass(type).equals(baseClass)) {
629             if (type instanceof Class) {
630                 // there is no useful information for us in raw types, so just
631                 // keep going.
632                 type = ((Class) type).getGenericSuperclass();
633             } else {
634                 ParameterizedType parameterizedType = (ParameterizedType) type;
635                 Class<?> rawType = (Class) parameterizedType.getRawType();
636 
637                 Type[] actualTypeArguments = parameterizedType.getActualTypeArguments();
638                 TypeVariable<?>[] typeParameters = rawType.getTypeParameters();
639                 for (int i = 0; i < actualTypeArguments.length; i++) {
640                     resolvedTypes.put(typeParameters[i], actualTypeArguments[i]);
641                 }
642 
643                 if (!rawType.equals(baseClass)) {
644                     type = rawType.getGenericSuperclass();
645                 }
646             }
647         }
648 
649         // finally, for each actual type argument provided to baseClass,
650         // determine (if possible)
651         // the raw class for that type argument.
652         Type[] actualTypeArguments;
653         if (type instanceof Class) {
654             actualTypeArguments = ((Class) type).getTypeParameters();
655         } else {
656             actualTypeArguments = ((ParameterizedType) type).getActualTypeArguments();
657         }
658         List<Class<?>> typeArgumentsAsClasses = new ArrayList<Class<?>>();
659         // resolve types by chasing down type variables.
660         for (Type baseType : actualTypeArguments) {
661             while (resolvedTypes.containsKey(baseType)) {
662                 baseType = resolvedTypes.get(baseType);
663             }
664             typeArgumentsAsClasses.add(getClass(baseType));
665         }
666         return typeArgumentsAsClasses;
667     }

670         Map<Type, Type> resolvedTypes = new HashMap<Type, Type>();
671         Type type = clazz;
