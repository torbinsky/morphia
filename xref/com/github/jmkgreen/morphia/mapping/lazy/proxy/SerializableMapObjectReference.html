25      public SerializableMapObjectReference(final Map mapToProxy, final Class referenceObjClass,
26                                            final boolean ignoreMissing, final DatastoreProvider p) {
27  
28          super(p, referenceObjClass, ignoreMissing);
29          object = mapToProxy;
30          keyMap = new LinkedHashMap<String, Key<?>>();
31      }
32  
33      public void __put(final String key, final Key k) {
34          keyMap.put(key, k);
35      }
36  
37      @Override
38      protected Object fetch() {
39          Map m = (Map) object;
40          m.clear();
41          // TODO us: change to getting them all at once and yell according to
42          // ignoreMissing in order to a) increase performance and b) resolve
43          // equals keys to the same instance
44          // that should really be done in datastore.
45          for (Map.Entry<?, Key<?>> e : keyMap.entrySet()) {
46              Key<?> entityKey = e.getValue();
47              Object entity = fetch(entityKey);
48              m.put(e.getKey(), entity);
49          }
50          return m;
51      }
52  
53      @Override
54      protected void beforeWriteObject() {
55          if (!__isFetched())
56              return;
57          else {
58              syncKeys();
59              ((Map) object).clear();
60          }
61      }
62  
63      private void syncKeys() {
64          Datastore ds = p.get();
65  
66          this.keyMap.clear();
67          Map<String, Object> map = (Map) object;
68          for (Map.Entry<String, Object> e : map.entrySet()) {
69              keyMap.put(e.getKey(), ds.getKey(e.getValue()));
70          }
71      }
72  
73      public Map<String, Key<?>> __getReferenceMap() {
74          return keyMap;
75      }
